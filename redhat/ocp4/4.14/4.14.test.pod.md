# build a testing container image

# build with docker file

```bash
# on vultr

mkdir -p /data/

cd /data

cat << EOF > /data/centos.test.dockerfile
FROM docker.io/rockylinux:9

RUN dnf -y install epel-release && dnf -y update
RUN dnf -y groupinstall 'Server' 'development' --allowerasing

RUN dnf -y install iperf3 tcpdump sysbench stress-ng fio nc python3 bpftool wireshark wireshark-cli siege

RUN dnf -y clean all 

RUN wget -O /usr/local/bin/bpftop https://github.com/Netflix/bpftop/releases/latest/download/bpftop && chmod +x /usr/local/bin/bpftop

EOF

podman build --squash -t quay.io/wangzheng422/qimgs:rocky9-test-2024.04.28 -f /data/centos.test.dockerfile ./

podman push quay.io/wangzheng422/qimgs:rocky9-test-2024.04.28

```
